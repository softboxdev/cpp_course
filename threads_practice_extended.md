```cpp
    else if (pid == 0) {
        // –î–æ—á–µ—Ä–Ω–∏–π –ø—Ä–æ—Ü–µ—Å—Å
        
        // –ó–∞–º–µ–Ω—è–µ–º –æ–±—Ä–∞–∑ –ø—Ä–æ—Ü–µ—Å—Å–∞ –Ω–æ–≤–æ–π –ø—Ä–æ–≥—Ä–∞–º–º–æ–π
        // execlp –∏—â–µ—Ç –ø—Ä–æ–≥—Ä–∞–º–º—É 'ls' –≤ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—è—Ö PATH
        // "ls" - –∏–º—è –ø—Ä–æ–≥—Ä–∞–º–º—ã –¥–ª—è –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è
        // "ls" - –ø–µ—Ä–≤—ã–π –∞—Ä–≥—É–º–µ–Ω—Ç (–æ–±—ã—á–Ω–æ –∏–º—è –ø—Ä–æ–≥—Ä–∞–º–º—ã)
        // "-l" - –≤—Ç–æ—Ä–æ–π –∞—Ä–≥—É–º–µ–Ω—Ç (–ø–æ–¥—Ä–æ–±–Ω—ã–π –≤—ã–≤–æ–¥)
        // "-a" - —Ç—Ä–µ—Ç–∏–π –∞—Ä–≥—É–º–µ–Ω—Ç (–ø–æ–∫–∞–∑–∞—Ç—å —Å–∫—Ä—ã—Ç—ã–µ —Ñ–∞–π–ª—ã)
        // NULL - –º–∞—Ä–∫–µ—Ä –∫–æ–Ω—Ü–∞ –∞—Ä–≥—É–º–µ–Ω—Ç–æ–≤
        execlp("ls", "ls", "-l", "-a", NULL);
        
        // –ï—Å–ª–∏ execlp –≤–µ—Ä–Ω—É–ª —É–ø—Ä–∞–≤–ª–µ–Ω–∏–µ, –∑–Ω–∞—á–∏—Ç –ø—Ä–æ–∏–∑–æ—à–ª–∞ –æ—à–∏–±–∫–∞
        // exec-—Ñ—É–Ω–∫—Ü–∏–∏ –≤–æ–∑–≤—Ä–∞—â–∞—é—Ç —É–ø—Ä–∞–≤–ª–µ–Ω–∏–µ —Ç–æ–ª—å–∫–æ –ø—Ä–∏ –æ—à–∏–±–∫–µ
        std::cerr << "–û—à–∏–±–∫–∞ exec: " << strerror(errno) << std::endl;
        exit(1);
    }
    else {
        // –†–æ–¥–∏—Ç–µ–ª—å—Å–∫–∏–π –ø—Ä–æ—Ü–µ—Å—Å
        std::cout << "–†–æ–¥–∏—Ç–µ–ª—å –∂–¥–µ—Ç –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è –¥–æ—á–µ—Ä–Ω–µ–≥–æ –ø—Ä–æ—Ü–µ—Å—Å–∞..." << std::endl;
        
        int status;
        pid_t finished_pid = wait(&status);
        
        if (finished_pid == -1) {
            std::cerr << "–û—à–∏–±–∫–∞ wait: " << strerror(errno) << std::endl;
        }
        else {
            if (WIFEXITED(status)) {
                // WIFEXITED –ø—Ä–æ–≤–µ—Ä—è–µ—Ç –Ω–æ—Ä–º–∞–ª—å–Ω–æ–µ –∑–∞–≤–µ—Ä—à–µ–Ω–∏–µ
                std::cout << "–ü—Ä–æ—Ü–µ—Å—Å " << finished_pid 
                          << " –∑–∞–≤–µ—Ä—à–∏–ª—Å—è —Å –∫–æ–¥–æ–º: " << WEXITSTATUS(status) << std::endl;
            }
            else if (WIFSIGNALED(status)) {
                // WIFSIGNALED –ø—Ä–æ–≤–µ—Ä—è–µ—Ç –∑–∞–≤–µ—Ä—à–µ–Ω–∏–µ –ø–æ —Å–∏–≥–Ω–∞–ª—É
                std::cout << "–ü—Ä–æ—Ü–µ—Å—Å " << finished_pid 
                          << " —É–±–∏—Ç —Å–∏–≥–Ω–∞–ª–æ–º: " << WTERMSIG(status) << std::endl;
            }
        }
    }

    return 0;
}
```

## 3. –ú–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω—ã–µ –ø—Ä–æ—Ü–µ—Å—Å—ã

### –°–æ–∑–¥–∞–Ω–∏–µ –Ω–µ—Å–∫–æ–ª—å–∫–∏—Ö –¥–æ—á–µ—Ä–Ω–∏—Ö –ø—Ä–æ—Ü–µ—Å—Å–æ–≤

```cpp
#include <unistd.h>
#include <sys/wait.h>
#include <vector>

int main() {

        else if (pid == 0) {
            // –î–æ—á–µ—Ä–Ω–∏–π –ø—Ä–æ—Ü–µ—Å—Å
            std::cout << "–î–æ—á–µ—Ä–Ω–∏–π –ø—Ä–æ—Ü–µ—Å—Å " << i 
                      << " —Å PID " << getpid() << " –∑–∞–ø—É—â–µ–Ω" << std::endl;
            
            // –ò–º–∏—Ç–∏—Ä—É–µ–º —Ä–∞–±–æ—Ç—É –ø—Ä–æ—Ü–µ—Å—Å–∞
            sleep(2 + i);  // –ö–∞–∂–¥—ã–π –ø—Ä–æ—Ü–µ—Å—Å —Å–ø–∏—Ç —Ä–∞–∑–Ω–æ–µ –≤—Ä–µ–º—è
            
            std::cout << "–î–æ—á–µ—Ä–Ω–∏–π –ø—Ä–æ—Ü–µ—Å—Å " << i << " –∑–∞–≤–µ—Ä—à–∞–µ—Ç—Å—è" << std::endl;
            exit(i);  // –í–æ–∑–≤—Ä–∞—â–∞–µ–º –Ω–æ–º–µ—Ä –ø—Ä–æ—Ü–µ—Å—Å–∞ –∫–∞–∫ –∫–æ–¥ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è
        }
        else {
            // –†–æ–¥–∏—Ç–µ–ª—å—Å–∫–∏–π –ø—Ä–æ—Ü–µ—Å—Å —Å–æ—Ö—Ä–∞–Ω—è–µ—Ç PID –¥–æ—á–µ—Ä–Ω–µ–≥–æ
            child_pids.push_back(pid);
        }
    }

    // –†–æ–¥–∏—Ç–µ–ª—å—Å–∫–∏–π –ø—Ä–æ—Ü–µ—Å—Å –æ–∂–∏–¥–∞–µ—Ç –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è –≤—Å–µ—Ö –¥–æ—á–µ—Ä–Ω–∏—Ö
    std::cout << "–†–æ–¥–∏—Ç–µ–ª—å—Å–∫–∏–π –ø—Ä–æ—Ü–µ—Å—Å (PID " << getpid() 
              << ") –æ–∂–∏–¥–∞–µ—Ç –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è " << child_pids.size() 
              << " –¥–æ—á–µ—Ä–Ω–∏—Ö –ø—Ä–æ—Ü–µ—Å—Å–æ–≤" << std::endl;

    for (size_t i = 0; i < child_pids.size(); ++i) {
        int status;
        // waitpid –æ–∂–∏–¥–∞–µ—Ç –∫–æ–Ω–∫—Ä–µ—Ç–Ω—ã–π –¥–æ—á–µ—Ä–Ω–∏–π –ø—Ä–æ—Ü–µ—Å—Å
        // child_pids[i] - PID –ø—Ä–æ—Ü–µ—Å—Å–∞, –∫–æ—Ç–æ—Ä—ã–π –Ω—É–∂–Ω–æ –∂–¥–∞—Ç—å
        // &status - —É–∫–∞–∑–∞—Ç–µ–ª—å –Ω–∞ –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é –¥–ª—è —Å—Ç–∞—Ç—É—Å–∞
        // 0 - –æ–ø—Ü–∏–∏ (0 –æ–∑–Ω–∞—á–∞–µ—Ç –±–µ–∑ —Å–ø–µ—Ü–∏–∞–ª—å–Ω—ã—Ö –æ–ø—Ü–∏–π)
        pid_t finished_pid = waitpid(child_pids[i], &status, 0);
        
        if (finished_pid == -1) {
            std::cerr << "–û—à–∏–±–∫–∞ –ø—Ä–∏ –æ–∂–∏–¥–∞–Ω–∏–∏ –ø—Ä–æ—Ü–µ—Å—Å–∞ " << child_pids[i] << std::endl;
        }

        }
    }

    std::cout << "–í—Å–µ –¥–æ—á–µ—Ä–Ω–∏–µ –ø—Ä–æ—Ü–µ—Å—Å—ã –∑–∞–≤–µ—Ä—à–µ–Ω—ã" << std::endl;
    return 0;
}
```

## 4. –û–±–º–µ–Ω –¥–∞–Ω–Ω—ã–º–∏ –º–µ–∂–¥—É –ø—Ä–æ—Ü–µ—Å—Å–∞–º–∏

### –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ pipe() –¥–ª—è –º–µ–∂–ø—Ä–æ—Ü–µ—Å—Å–Ω–æ–≥–æ –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è

```cpp

        std::cerr << "–û—à–∏–±–∫–∞ fork" << std::endl;
        return 1;

        write(pipefd[1], message.c_str(), message.length() + 1);
        
        // –ó–∞–∫—Ä—ã–≤–∞–µ–º –¥–µ—Å–∫—Ä–∏–ø—Ç–æ—Ä –∑–∞–ø–∏—Å–∏
        close(pipefd[1]);
        
        exit(0);
    }
    else {
        // –†–æ–¥–∏—Ç–µ–ª—å—Å–∫–∏–π –ø—Ä–æ—Ü–µ—Å—Å - –±—É–¥–µ—Ç —á–∏—Ç–∞—Ç—å –∏–∑ pipe
        
        // –ó–∞–∫—Ä—ã–≤–∞–µ–º –Ω–µ–Ω—É–∂–Ω—ã–π –¥–µ—Å–∫—Ä–∏–ø—Ç–æ—Ä –¥–ª—è –∑–∞–ø–∏—Å–∏

    }

    return 0;
```
